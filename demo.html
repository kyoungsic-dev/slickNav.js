slickNav.js는 kenwheeler의 slick.js에 커스텀 네비게이션을 추가한 확장 Plugin 입니다.

  • [slickNav.js 적용 전 기본 네비게이션]

    웹 접근성을 고려한 play/pause 버튼 및 UI가 제공되지 않는 불편함이 있습니다. 

      
$('.slick-default').slick({
  dots: true,
  arrows: true,
  infinite: true,
  speed: 250,
  slidesToShow: 1,
  slidesToScroll: 1,
  autoplay: true,
  autoplaySpeed: 5000,
  cssEase: 'linear',
  slide: '.item', // 자식 요소 안의 실제 슬라이드 되는 Node를 지정해주는 것이 좋습니다.
  pauseOnHover: false,
  pauseOnFocus: false,
})

  • [slickNav.js 적용 전 slick.js 적용 (공통)]

      
$('.slick-nav').slick({
  dots: false, // 기본 dots false
  arrows: false, // 기본 arrows false
  infinite: true,
  speed: 250,
  slidesToShow: 1,
  slidesToScroll: 1,
  autoplay: true,
  autoplaySpeed: 5000,
  cssEase: 'linear',
  slide: '.item',
  pauseOnHover: false,
  pauseOnFocus: false,
})

  • [slickNav.js 전체 옵션]

      
$('.slick-nav').slickNav('.item', { // 실제 slide 되는 Node를 select
  circle: false, // [Circle 애니메이션 사용 유무] 기본값 : false. 아래 속성을 이용해 간단한 스타일 제어   * false 시 Dot만 출력
  circleSize: '32px', // [Circle 사이즈] 기본값 : 30px
  circleBorder: '2px', // [Circle 테두리 굵기] 기본값: 1px
  circleTrack: 'rgba(0, 0, 0, .2)', // [Circle 테두리 배경색] 기본값 :  rgba(0, 0, 0, .2)

  progress: true, // [프로그레스 바 사용 유무] *Circle과 동시 사용 불가 
  progressWidth: '150px', // [프로그레스 바 width] 기본값 : 150px 
  progressHeight: '2px', // [프로그레스 바 height] 기본값 : 3px
  progressTrack: '#000', // [프로그레스 바 배경색] 기본값 : #000
  progressFill: '#f00', // [프로그레스 바 채움색] 기본값 : #fff
  
  dotSize: '6px', //  [Dot 크기] 기본값 : 5px
  dotColor: '#3598db', // [Dot 활성화 색상] 기본값 : #f00
  dotDefaultColor: '#ccc', // [Dot 기본 색상] 기본값 : #ccc


  controlOnly: false, // [prev, pause, next 버튼만 출력] 기본값 : false
                    
  duration: 5000, // 슬라이드 autoplay 시간과 동일한 값
});
