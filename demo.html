slickNav.js

slickNav.js는 kenwheeler의 slick.js Plugin에 커스텀 네비게이션을 추가한 확장 Plugin 입니다.

  • slickNav.js 적용 전

    웹 접근성을 고려한 play/pause 버튼 및 네비게이션의 UI가 제공되지 않습니다.

    Javascript

    
  $('.slick-default').slick({
    dots: true,
    arrows: true,
    infinite: true,
    speed: 250,
    slidesToShow: 1,
    slidesToScroll: 1,
    autoplay: true,
    autoplaySpeed: 5000,
    cssEase: 'linear',
    slide: '.item', // 자식 요소 안의 실제 슬라이드 되는 Node를 지정해주는 것이 좋습니다.
    pauseOnHover: false,
    pauseOnFocus: false,
  })

  • 사용방법

    HTML

    
  <!-- import slickNav.js -->
  <script defer src="/js/slickNav.js"></script>

    Javascript

    
  // 전체 옵션
  $('.slick-nav').slickNav('.item', { // 실제 slide 되는 Node를 select
    parentNode: '#parent-tag', // [append 시킬 부모 노드] 기본값 : false. * 값이 false 일 때 initialize 된 slick 노드가 부모 노드 (예시에서는 .slick-nav)
    
    circle: false, // [Circle 애니메이션 사용 유무] 기본값 : false. 아래 속성을 이용해 간단한 스타일 제어   * false 시 Dot만 출력
    circleSize: '32px', // [Circle 사이즈] 기본값 : 30px
    circleBorder: '2px', // [Circle 테두리 굵기] 기본값: 1px
    circleTrack: 'rgba(0, 0, 0, .2)', // [Circle 테두리 배경색] 기본값 :  rgba(0, 0, 0, .2)

    progress: true, // [프로그레스 바 사용 유무] *Circle과 동시 사용 불가 
    progressWidth: '150px', // [프로그레스 바 width] 기본값 : 150px 
    progressHeight: '2px', // [프로그레스 바 height] 기본값 : 3px
    progressTrack: '#000', // [프로그레스 바 배경색] 기본값 : #000
    progressFill: '#276ff7', // [프로그레스 바 채움색] 기본값 : #fff
    
    dotSize: '6px', //  [Dot 크기] 기본값 : 5px
    dotColor: '#276ff7', // [Dot, Circle 활성화 색상] 기본값 : #f00
    dotDefaultColor: '#ccc', // [Dot, Circle 기본 색상] 기본값 : #ccc


    controls: true, // [prev, pause/play, next 버튼 노출 여부 (true 시 노출)] 기본값 : true
    controlsOnly: false, // [prev, pause/play, next 버튼만 출력] 기본값 : false
                      
    duration: 5000, // 슬라이드 autoplay 시간과 동일한 값
  });

  • [적용예시01] dots 및 play/pause 버튼

    Javascript

    
  // dots 및 play/pause 버튼
  $('.slick-nav').slickNav('.item',{
    dotSize: '6px', //  [Dot 크기] 기본값 : 5px
    dotColor: '#276ff7', // [Dot 활성화 색상] 기본값 : #f00
    dotDefaultColor: '#ccc', // [Dot 기본 색상] 기본값 : #ccc
    duration: 5000, // 슬라이드 autoplay 시간과 동일한 값
    controls: false, // [prev, pause/play, next 버튼 노출 여부 (true 시 노출)] 기본값 : true
  });
